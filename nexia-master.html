} exports.handler = async (event) => { const CORS = { 'Access-Control-Allow-Origin': '*', 'Access-Control-Allow-Methods': 'POST, GET, OPTIONS', 'Access-Control-Allow-Headers': 'Content-Type' }; if (event.httpMethod === 'OPTIONS') return { statusCode: 200, headers: CORS, body: '' }; const ticketId = generateTicketId(); const isCes = tenantId.includes('CES'); const qrPayload = Buffer.from(JSON.stringify({ t: ticketId, e: eventId, n: tenantId, v: '1' })).toString('base64'); const qrUrl = `https://nexiaos.netlify.app/ces/checkin.html?tk=${qrPayload}`; if (t.checkedIn) return { statusCode: 200, headers: CORS, body: JSON.stringify({ valid: false, reason: 'Check-in já realizado', holderName: t.holderName }) }; await ref.update({ checkedIn: true, checkinAt: ts, status: 'used' }); return { statusCode: 200, headers: CORS, body: JSON.stringify({ valid: true, ticketId: inTicketId, holderName: t.holderName, holderEmail: t.holderEmail, packageName: t.packageName }) }; } if (action === 'list') { const snap = await tenantRef.collection('tickets').where('eventId', '==', eventId).orderBy('issuedAt', 'desc').limit(100).get(); return { statusCode: 200, headers: CORS, body: JSON.stringify({ tickets: snap.docs.map(d => ({ id: d.id, ...d.data() })), count: snap.size }) }; } return { statusCode: 400, headers: CORS, body: JSON.stringify({ error: `Unknown action: ${action}` }) }; } catch (err) { console.error('[ticket-engine]', err); return { statusCode: 500, headers: CORS, body: JSON.stringify({ error: err.message }) }; } };
